function calculate() { let A = parseFloat(document.getElementById('initialA').value) || 0; let B = parseFloat(document.getElementById('annualB').value) || 0; let C = parseFloat(document.getElementById('annualC').value) / 100 || 0; let D = parseFloat(document.getElementById('finalD').value) || 0; let E = parseFloat(document.getElementById('yearsE').value) || 0; // 檢查哪個變數沒有輸入，並進行相應的計算 if (!D && A && B && C && E) { D = A * Math.pow((1 + C), E) + B * ((Math.pow((1 + C), E) - 1) / C); document.getElementById('result').innerText = "期末終值 D 是: " + D.toFixed(2); } else if (!A && D && B && C && E) { A = (D - B * ((Math.pow((1 + C), E) - 1) / C)) / Math.pow((1 + C), E); document.getElementById('result').innerText = "起始資金 A 是: " + A.toFixed(2); } else if (!B && D && A && C && E) { B = (D - A * Math.pow((1 + C), E)) / ((Math.pow((1 + C), E) - 1) / C); document.getElementById('result').innerText = "每年投入 B 是: " + B.toFixed(2); } else if (!C && D && A && B && E) { C = Math.pow((D - B * ((Math.pow((1 + C), E) - 1) / C)) / A, 1 / E) - 1; document.getElementById('result').innerText = "年化報酬率 C 是: " + (C * 100).toFixed(2) + "%"; } else if (!E && D && A && B && C) { E = Math.log(D / (A + B * ((Math.pow((1 + C), E) - 1) / C))) / Math.log(1 + C); document.getElementById('result').innerText = "經過時間 E 是: " + E.toFixed(2) + " 年"; } else { document.getElementById('result').innerText = "請輸入至少四個變數"; } }
