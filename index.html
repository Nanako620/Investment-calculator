function calculate() { let A = parseFloat(document.getElementById('initialA').value); let B = parseFloat(document.getElementById('annualB').value); let C = parseFloat(document.getElementById('annualC').value) / 100; let D = parseFloat(document.getElementById('finalD').value); let E = parseFloat(document.getElementById('yearsE').value); // 計算期末終值 D if (!D && A && B && C && E) { D = A * Math.pow((1 + C), E) + B * ((Math.pow((1 + C), E) - 1) / C); document.getElementById('result').innerText = "期末終值 D 是: " + D.toFixed(2); } // 計算起始資金 A else if (!A && D && B && C && E) { A = (D - B * ((Math.pow((1 + C), E) - 1) / C)) / Math.pow((1 + C), E); document.getElementById('result').innerText = "起始資金 A 是: " + A.toFixed(2); } // 計算每年投入 B else if (!B && D && A && C && E) { B = (D - A * Math.pow((1 + C), E)) / ((Math.pow((1 + C), E) - 1) / C); document.getElementById('result').innerText = "每年投入 B 是: " + B.toFixed(2); } // 計算年化報酬率 C else if (!C && D && A && B && E) { C = Math.pow(D / (A + B * ((Math.pow((1 + C), E) - 1) / C))), 1 / E) - 1; document.getElementById('result').innerText = "年化報酬率 C 是: " + (C * 100).toFixed(2) + "%"; } // 計算經過時間 E else if (!E && D && A && B && C) { E = Math.log(D / (A + B * ((Math.pow((1 + C), E) - 1) / C))) / Math.log(1 + C); document.getElementById('result').innerText = "經過時間 E 是: " + E.toFixed(2) + " 年"; } else { document.getElementById('result').innerText = "請輸入至少四個變數。"; } }
